protocol UDP {
    meta {
        endian = big;
        mode   = datagram;
    }

    message Datagram {
        src_port: u16;
        dst_port: u16;
        length:   u16;
        validate length >= 8 -> "UDP length must include 8-byte header";
        checksum: u16;
        payload:  bytes[length - 8];
    }
}
